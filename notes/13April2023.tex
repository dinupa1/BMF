  \begin{verbatim}
  4.5 < mass && mass < 8.8
  D < 200.

  \begin{verbatim}
  mass = {2, 4.5, 6.5};
  pT = {3, 0., 1.5};
  xF = {4, 0., 0.8};
  phi = {20, -pi, pi};
  costh = {20, -0.5, 0.5};
  \end{verbatim}
